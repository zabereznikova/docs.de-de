## <a name="example"></a><span data-ttu-id="48242-122">Beispiel</span><span class="sxs-lookup"><span data-stu-id="48242-122">Example</span></span>

```csharp
enum E
{
    A,
    B
}

class C
{
    // Code with violations
    int M(E e)
    {
        // IDE0010: Add missing cases
        switch (e)
        {
            case E.A:
                return 0;
        }

        return -1;
    }

    // Fixed code
    int M(E e)
    {
        switch (e)
        {
            case E.A:
                return 0;
            case E.B:
                return 1;
            default:
                return -1;
        }
    }
}
```
