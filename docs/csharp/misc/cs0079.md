# <a name="compiler-error-cs0079"></a>Compilerfehler CS0079
Das Ereignis "Ereignis" kann nur links von += oder -= verwendet werden  
  
 Ein [Ereignis](../language-reference/keywords/event.md) wurde falsch aufgerufen. Weitere Informationen finden Sie unter [Ereignisse](../programming-guide/events/index.md) und [Delegaten](../programming-guide/delegates/index.md).  
  
 Im folgenden Beispiel wird CS0079 generiert:  
  
```csharp  
// CS0079.cs  
using System;  
  
public delegate void MyEventHandler();  
  
public class Class1  
{  
   private MyEventHandler _e;  
  
   public event MyEventHandler Pow  
   {  
      add  
      {  
         _e += value;  
         Console.WriteLine("in add accessor");  
      }  
      remove  
      {  
         _e -= value;  
         Console.WriteLine("in remove accessor");  
      }  
   }  
  
   public void Handler()  
   {  
   }  
  
   public void Fire()  
   {  
      if (_e != null)  
      {  
         Pow();   // CS0079  
         // try the following line instead  
         // _e();  
      }  
   }  
  
   public static void Main()  
   {  
      Class1 p = new Class1();  
      p.Pow += new MyEventHandler(p.Handler);  
      p._e();  
      p.Pow += new MyEventHandler(p.Handler);  
      p._e();  
      p._e -= new MyEventHandler(p.Handler);  
      if (p._e != null)  
      {  
         p._e();  
      }  
      p.Pow -= new MyEventHandler(p.Handler);  
      if (p._e != null)  
      {  
         p._e();  
      }  
   }  
}  
```
