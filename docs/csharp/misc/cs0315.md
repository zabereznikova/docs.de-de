## <a name="example"></a><span data-ttu-id="b7881-108">Beispiel</span><span class="sxs-lookup"><span data-stu-id="b7881-108">Example</span></span>

// cs0315.cs

public class Gen<T> where T : ClassConstraint  
{         
}
